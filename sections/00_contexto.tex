\begin{frame}{¿Qué es un camino?}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Definición — camino}
        Un \textbf{camino} en $X$ es una función continua
        \[
          \alpha : \I \longrightarrow X.
        \]
        $\alpha(0)$ es el \emph{punto inicial} y $\alpha(1)$ el
        \emph{punto terminal}.
      \end{block}
      \medskip
      Pensando en $t$ como \emph{tiempo}, un punto se mueve desde
      $\alpha(0)$ y traza una ruta continua hasta $\alpha(1)$.
    \end{column}
    \begin{column}{0.5\textwidth}
      \centering
      \begin{tikzpicture}[scale=1.0]
        % Espacio X
        \draw[thick, fill=grisClaro]
          plot[smooth cycle, tension=0.7]
          coordinates {(-2.4,0.2) (-1.7,1.5) (0,2.0) (1.7,1.5) (2.4,0.2)
                       (1.8,-1.2) (0,-1.7) (-1.8,-1.2)};
        \node at (1.7,1.55) {\small $X$};

        % Camino alpha
        \draw[morado, very thick, -Stealth]
          (-1.5,-0.6)
            .. controls (-0.8,0.6) and (0.3,-0.4) .. (1.4,0.7);
        % Puntos inicial / terminal
        \filldraw[morado] (-1.5,-0.6) circle (1.8pt)
          node[below left, morado]{\small $\alpha(0)$};
        \filldraw[morado] (1.4,0.7) circle (1.8pt)
          node[above right, morado]{\small $\alpha(1)$};
        % Puntos intermedios (punto en movimiento)
        \foreach \p in {(-0.95,0.16), (-0.2,0.06), (0.55,0.05)}{
          \filldraw[morado!55] \p circle (1.1pt);
        }
        \node[morado] at (-0.2,0.85) {\small $\alpha$};
      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Caminos equivalentes — homotopía de caminos}
  Dos caminos con \emph{los mismos extremos} son \textbf{equivalentes} si uno
  se deforma continuamente en el otro mediante $H:\IxI\to X$ con
  $H(\cdot,0)=\alpha$, $H(\cdot,1)=\beta$ y extremos fijos.

  \bigskip
  \begin{center}
  \begin{tikzpicture}[scale=1.0]
    % Extremos comunes
    \filldraw[ambar] (-3.2,0) circle (2pt)
      node[left, ambar]{\small $\alpha(0)=\beta(0)$};
    \filldraw[ambar] (3.2,0) circle (2pt)
      node[right, ambar]{\small $\alpha(1)=\beta(1)$};

    % beta (arriba)
    \draw[verdeTeal, very thick]
      (-3.2,0) .. controls (-1.5,1.6) and (1.5,1.6) .. (3.2,0);
    \node[verdeTeal] at (0,1.55) {\small $\beta$};
    % nivel intermedio (punteado)
    \draw[coralOscuro, thick, dashed]
      (-3.2,0) .. controls (-1.5,0.7) and (1.5,0.7) .. (3.2,0);
    \node[coralOscuro] at (1.7,0.78) {\small $H(\cdot,\tfrac{1}{2})$};
    % alpha (abajo)
    \draw[morado, very thick]
      (-3.2,0) .. controls (-1.5,-1.2) and (1.5,-1.2) .. (3.2,0);
    \node[morado] at (0,-1.2) {\small $\alpha$};

    % flechas de deformación
    \foreach \x in {-1.6,0,1.6}{
      \draw[-Stealth, gray] (\x,-0.55) -- (\x,1.0);
    }
  \end{tikzpicture}
  \end{center}

  \medskip
  \centering
  \small La homotopía $H$ \emph{empuja} $\alpha$ hasta $\beta$ sin mover los
  extremos. Si hubiera un \emph{agujero} entre ellos, no serían equivalentes.
\end{frame}

\begin{frame}{Lazos y punto base}
  \begin{columns}[T]
    \begin{column}{0.46\textwidth}
      \begin{block}{Definición — lazo}
        Un \textbf{lazo} con base $\xo$ es un camino con
        \[
          \alpha(0) = \alpha(1) = \xo .
        \]
      \end{block}
      \medskip
      \begin{itemize}
        \item Sale de $\xo$ y regresa a $\xo$.
        \item $\pione(X,\xo)$ reúne las \emph{clases de homotopía}
              de todos los lazos con base $\xo$.
      \end{itemize}
    \end{column}
    \begin{column}{0.5\textwidth}

\begin{frame}{Producto de caminos $\alpha * \beta$}
  Si $\alpha(1)=\beta(0)$, el \textbf{producto} recorre $\alpha$ al doble de
  velocidad y luego $\beta$ al doble de velocidad:
  \[
    (\alpha * \beta)(t) =
    \begin{cases}
      \alpha(2t)    & 0 \le t \le \tfrac{1}{2} \\[3pt]
      \beta(2t-1)   & \tfrac{1}{2} \le t \le 1
    \end{cases}
  \]

  \bigskip
  \begin{center}
  \begin{tikzpicture}[scale=1.0]
    % Línea de tiempo
    \draw[thick, -Stealth] (-0.2,0) -- (7.2,0) node[right]{\small $t$};
    % Bloque alpha
    \fill[moradoclaro!60] (0,0.15) rectangle (3.5,0.55);
    \node at (1.75, 0.35) {\small $\alpha$ (doble velocidad)};
    % Bloque beta
    \fill[verdeTeal!20] (3.5,0.15) rectangle (7,0.55);
    \node at (5.25, 0.35) {\small $\beta$ (doble velocidad)};
    % Marcas
    \foreach \x/\lab in {0/{$0$}, 3.5/{$\tfrac{1}{2}$}, 7/{$1$}}{
      \draw (\x, -0.12) -- (\x, 0.12);
      \node[below] at (\x, -0.12) {\small \lab};
    }
    % Puntos
    \fill[ambar] (0,0) circle (3pt) node[below=16pt, ambar]{\tiny $\alpha(0)$};
    \fill[ambar] (3.5,0) circle (3pt);
    \fill[ambar] (7,0) circle (3pt) node[below=16pt, ambar]{\tiny $\beta(1)$};
    \node[above=4pt, ambar] at (3.5, 0.55) {\tiny $\alpha(1)=\beta(0)$};
  \end{tikzpicture}
  \end{center}

  \medskip
  \centering
  \small La condición $\alpha(1)=\beta(0)$ evita discontinuidades en
  $t=\tfrac{1}{2}$ (Lema de Continuidad).
\end{frame}

              \textcolor{coralOscuro}{$C_2$}: no se puede cruzar el agujero.
      \end{itemize}

\begin{frame}{¿Qué estamos construyendo?}
  \begin{block}{Definición — $\pi_1(X,\xo)$}
    El \textbf{grupo fundamental} de $X$ en $\xo$ es el conjunto de clases de
    homotopía de lazos con base $\xo$, con la operación
    \[
      \clase{\alpha} \circ \clase{\beta} \;=\; \clase{\alpha * \beta}
    \]
    donde $(\alpha * \beta)(t) =
      \begin{cases} \alpha(2t) & 0 \le t \le \tfrac{1}{2}\\
                    \beta(2t-1) & \tfrac{1}{2} \le t \le 1 \end{cases}$
  \end{block}

  \bigskip
  \textbf{Teorema 4.2:} $\pione(X,\xo)$ es un \emph{grupo} bajo $\circ$.
  \medskip

\begin{frame}{Herramienta clave: Lema de Continuidad}
  \begin{alertblock}{Lema de Continuidad}
    Sea $X = A \cup B$ con $A, B$ cerrados. Si $f:A \to Y$ y $g:B \to Y$ son
    continuas con $f(x)=g(x)$ para todo $x \in A \cap B$, entonces
    \[
      h(x) = \begin{cases} f(x) & x \in A \\ g(x) & x \in B \end{cases}
    \]
    es continua.
  \end{alertblock}

  \bigskip
  \textbf{Uso práctico en este capítulo:} todas las homotopías se definen por
  tramos sobre el cuadrado $\IxI$.  El Lema garantiza la continuidad cuando
  las dos piezas coinciden en la frontera común.
\end{frame}
