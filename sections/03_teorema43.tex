% ============================================================
\section{Teorema 4.3 — Independencia del punto base}
% ============================================================

\begin{frame}{Teorema 4.3 — Enunciado}

  \begin{block}{Teorema 4.3}
    Si $X$ es un espacio arco-conexo y $\xo, x_1 \in X$, entonces
    \[
      \pione(X, \xo) \;\cong\; \pione(X, x_1).
    \]
  \end{block}

  \bigskip
  \textbf{Idea central:} cualquier camino $\rho$ de $\xo$ a $x_1$ induce
  un isomorfismo por \emph{conjugación}:
  \[
    [\alpha] \;\longmapsto\; [\bar{\rho} * \alpha * \rho].
  \]

  \bigskip
  \begin{itemize}
    \item El resultado justifica hablar de \emph{el} grupo fundamental
          $\pione(X)$ de un espacio arco-conexo (como grupo abstracto).
    \item \textbf{Cuidado:} el isomorfismo \emph{depende} del camino $\rho$
          elegido; no hay un isomorfismo canónico en general.
  \end{itemize}

\end{frame}

\begin{frame}{Teorema 4.3 — Construcción de $P$}

  Sea $\rho : I \to X$ un camino con $\rho(0) = \xo$ y $\rho(1) = x_1$.

  \medskip
  \begin{block}{Definición de $P$}
    \[
      P : \pione(X,\xo) \to \pione(X,x_1),
      \qquad
      \boxed{P([\alpha]) \;=\; [(\bar{\rho} * \alpha) * \rho].}
    \]
  \end{block}

  \medskip
  \textbf{$(\bar{\rho} * \alpha) * \rho$ es lazo en $x_1$:}
  \begin{itemize}
    \item $\bar{\rho}$ va de $x_1$ a $\xo$; $\alpha$ es lazo en $\xo$;
          $\rho$ va de $\xo$ a $x_1$. La concatenación empieza y
          termina en $x_1$. \hfill $\checkmark$
  \end{itemize}

  \medskip
  \textbf{$P$ está bien definida:} si $\alpha \simxo \alpha'$ mediante
  una homotopía de lazos $F : I \times I \to X$, entonces
  \[
    (\bar{\rho} * F(\cdot, s)) * \rho
  \]
  es una homotopía de lazos de $\bar{\rho} * \alpha * \rho$ a
  $\bar{\rho} * \alpha' * \rho$ en $x_1$. Luego $P([\alpha]) = P([\alpha'])$.
  \hfill $\checkmark$

\end{frame}

\begin{frame}{Teorema 4.3 — $P$ es homomorfismo}

  Sean $[\alpha], [\beta] \in \pione(X, \xo)$.

  \begin{exampleblock}{Verificación}
    \begin{align*}
      P\!\left([\alpha]\cdot[\beta]\right)
        &= P([\alpha * \beta])
         = [\bar{\rho} * \alpha * \beta * \rho] \\[4pt]
        &= [\bar{\rho} * \alpha * \rho * \bar{\rho} * \beta * \rho]
    \end{align*}
  \end{exampleblock}

  \medskip
  \textbf{Paso clave:} se inserta $\rho * \bar{\rho}$ (homotópico al lazo
  constante $c_{\xo}$ por el Lema B) entre $\alpha$ y $\beta$.
  El Lema C garantiza que podemos reasociar libremente, luego:
  \[
    = [\bar{\rho}*\alpha*\rho]\cdot[\bar{\rho}*\beta*\rho]
    = P([\alpha])\cdot P([\beta]).
  \]

  Por tanto $P$ es un homomorfismo de grupos. \hfill $\checkmark$

\end{frame}

\begin{frame}{Teorema 4.3 — El inverso $Q$: $QP = \mathrm{id}$}

  \begin{block}{Definición de $Q$}
    \[
      Q : \pione(X,x_1) \to \pione(X,\xo),
      \qquad
      Q([\sigma]) \;=\; [\rho * \sigma * \bar{\rho}].
    \]
  \end{block}

  \medskip
  \textbf{$QP = \mathrm{id}_{\pione(X,\xo)}$:} Sea $[\alpha] \in \pione(X,\xo)$.
  \begin{align*}
    QP([\alpha])
      &= Q\!\left([\bar{\rho}*\alpha*\rho]\right) \\
      &= [\rho * \bar{\rho} * \alpha * \rho * \bar{\rho}] \\
      &= [\rho*\bar{\rho}] \cdot [\alpha] \cdot [\rho*\bar{\rho}].
  \end{align*}

  Por el Lema B, $\rho * \bar{\rho} \simxo c_{\xo}$, y el lazo constante
  es el elemento neutro:
  \[
    = [c_{\xo}] \cdot [\alpha] \cdot [c_{\xo}] \;=\; [\alpha].
  \]
  \hfill $\checkmark$

\end{frame}

\begin{frame}{Teorema 4.3 — $PQ = \mathrm{id}$ y conclusión}

  \textbf{$PQ = \mathrm{id}_{\pione(X,x_1)}$:} Por simetría, intercambiamos
  los papeles de $\xo$ y $x_1$ (y de $\rho$ con $\bar{\rho}$).
  Sea $[\sigma] \in \pione(X,x_1)$:
  \begin{align*}
    PQ([\sigma])
      &= P\!\left([\rho*\sigma*\bar{\rho}]\right) \\
      &= [\bar{\rho}*\rho*\sigma*\bar{\rho}*\rho] \\
      &= [\bar{\rho}*\rho] \cdot [\sigma] \cdot [\bar{\rho}*\rho].
  \end{align*}

  Ahora $\bar{\rho}*\rho \sim_{x_1} c_{x_1}$ por el Lema B:
  \[
    = [c_{x_1}] \cdot [\sigma] \cdot [c_{x_1}] \;=\; [\sigma].
  \]
  \hfill $\checkmark$

  \begin{block}{Conclusión}
    $Q \circ P = \mathrm{id}$ y $P \circ Q = \mathrm{id}$, luego $P$ es
    biyectiva. Como también es homomorfismo:
    \[
      P : \pione(X, \xo) \;\xrightarrow{\;\cong\;}\; \pione(X, x_1).
    \]
    \hfill $\square$
  \end{block}

\end{frame}

\begin{frame}{Teorema 4.3 — Observación sobre el punto base}

  \begin{exampleblock}{Corolario}
    Para un espacio arco-conexo $X$, todos los grupos $\pione(X,x)$
    son isomorfos. Se puede hablar de \emph{el} grupo fundamental
    $\pione(X)$ como grupo abstracto.
  \end{exampleblock}

  \bigskip
  \begin{alertblock}{Advertencia: el isomorfismo no es único}
    \begin{itemize}
      \item Diferentes caminos $\rho$ de $\xo$ a $x_1$ producen, en general,
            \textbf{diferentes isomorfismos} $P_\rho$.
      \item $P_\rho$ y $P_{\rho'}$ difieren por un automorfismo interior de
            $\pione(X,x_1)$ (conjugación por $[\bar{\rho}' * \rho]$).
      \item Al comparar $\pione(X,\xo)$ con $\pione(Y,f(\xo))$ a través de
            $f : X \to Y$, el punto base es \textbf{indispensable}.
    \end{itemize}
  \end{alertblock}

\end{frame}
